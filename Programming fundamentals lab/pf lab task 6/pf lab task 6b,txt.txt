#include<iostream>
using namespace std;
int main()

{
	char u='y';
	while(u=='y')
	{
	int x , y;
	char opr;
	cout<<"enter first number =";
	cin>>x;
	cout<<"enter second number =";
	cin>>y;
	cout<<"enter the operation you want to perform =";
	cin>>opr;
	if (opr='+')
	{
	cout<<"sum ="<<x+y<<endl;
	}
	else if (opr='-')
	{
	cout<<"subtraction ="<<x-y<<endl;
	}
	else if (opr='/')
	{
	cout<<"division ="<<x/y<<endl;
	}
	else if (opr='*')
	{
	cout<<"multiplication ="<<x*y<<endl;
	}
	else
	{
	cout<<"wrong key"<<endl;
	}
	cout<<"do you want to continue (y/n)?";
	cin>>u;
	}
	system ("pause");
	return 0;
}
